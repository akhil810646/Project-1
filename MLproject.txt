  <script>
    document.getElementById("predictionForm").addEventListener("submit", function (e) {
      e.preventDefault();

      const weight = parseFloat(document.getElementById("weight").value);
      const energy = parseFloat(document.getElementById("energy").value);
      const fertilizer = parseFloat(document.getElementById("fertilizer").value);
      const co2 = parseFloat(document.getElementById("co2").value);

      // Simple placeholder calculation
      const green = (weight * 80 + energy * 0.4).toFixed(2);
      const blue = (fertilizer * 3 + co2 * 0.3).toFixed(2);
      const grey = (weight + energy + fertilizer + co2).toFixed(2);

      document.getElementById("greenResult").textContent = green;
      document.getElementById("blueResult").textContent = blue;
      document.getElementById("greyResult").textContent = grey;

      const results = document.getElementById("results");
      results.classList.add("visible");

      results.scrollIntoView({ behavior: "smooth" });
    });
